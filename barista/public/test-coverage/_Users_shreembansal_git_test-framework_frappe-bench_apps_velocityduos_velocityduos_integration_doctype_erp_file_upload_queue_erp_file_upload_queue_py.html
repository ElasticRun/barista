from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

 

from velocityduos.integration.erp_client_login import login_erp_client,get_erp_url 

from velocityduos.api.utilities import get_attachments 

 

import urllib3 

urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning) 

 

headers = {'content-type': 'application/x-www-form-urlencoded'} 

 

class ERPFileUploadQueue(Document): 

pass 

 

def upload_pending_files(): 

pending_files = frappe.get_all("ERP File Upload Queue",filters={'status':'Pending Upload'},limit_page_length=10) 

if len(pending_files) > 0 : 

session = login_erp_client() 

session.headers = headers 

pending_files = [d.name for d in pending_files] 

frappe.db.sql("""update `tabERP File Upload Queue` set status='Uploading' where name in %(pending_files)s""",{'pending_files':pending_files},auto_commit=1) 

url = get_erp_url() 

for files in pending_files: 

doc = frappe.get_doc("ERP File Upload Queue",files) 

try: 

try: 

if not session or session.get("frappe.auth.get_logged_user").status_code != 200: 

raise frappe.ValidationError("Login ERP Client") 

except Exception: 

session = login_erp_client() 

 

upload_file_attachment(doc.source_doctype,doc.source_name,doc.erp_doctype, doc.erp_docname,session,url) 

doc.status = 'Uploaded' 

except Exception: 

frappe.log_error(frappe.get_traceback(),'erp_file_upload_queue-{0}'.format(doc.source_name)) 

doc.logs = frappe.get_traceback() 

doc.status = 'Error' 

doc.save(True) 

 

def upload_file_attachment(vel_doctype,vel_docname,erp_doctype,erp_docname,session,url,retry=False): 

# Get Attachments for Document 

attachments = get_attachments(vel_doctype,vel_docname) 

if len(attachments) > 0: 

for file_doc in attachments: 

file_content = frappe.utils.file_manager.get_file(file_doc.file_url) 

file_data = None 

if type(file_content[1]) is str: 

file_data = frappe.utils.file_manager.base64.b64encode(file_content[1].encode('utf-8')) 

else: 

file_data = frappe.utils.file_manager.base64.b64encode(file_content[1]) 

file_name = file_doc.file_name if file_doc.file_name else file_doc.file_url.split("/")[-1] 

data = {'cmd':'uploadfile','doctype':erp_doctype,'docname':erp_docname, 'is_private':'0','filename': file_name, 'filedata': file_data,'from_form':1} 

res = session.post(url,data) 

if res.status_code != 200: 

if retry: 

session = login_erp_client() 

res = session.post(url,data) 

if res.status_code != 200: 

frappe.throw("{0}\n{1}\n{2}".format(file_name,frappe.get_traceback(),res.content)) 

else: 

frappe.throw("{0}\n{1}\n{2}".format(file_name,frappe.get_traceback(),res.content)) 

 

def retry_failed_uploads(): 

pending_files = frappe.get_all("ERP File Upload Queue",filters={'status':'Error'},limit_page_length=10) 

pending_files = [d.name for d in pending_files] 

 

session = login_erp_client() 

headers = {'content-type': 'application/x-www-form-urlencoded'} 

session.headers = headers 

url = get_erp_url() 

 

for files in pending_files: 

doc = frappe.get_doc("ERP File Upload Queue",files) 

try: 

upload_file_attachment(doc.source_doctype,doc.source_name,doc.matrix_doctype, doc.matrix_docname,session,url,retry=True) 

doc.status = 'Uploaded' 

doc.save(True) 

frappe.db.commit() 

except Exception: 

print("{0}-{1}".format(doc.name,frappe.get_traceback()))
