import frappe 

import json 

from velocityduos.integration.erp_client_login import login_erp_client,get_erp_url 

 

def fetch_erp_project(): 

session = login_erp_client() 

url = get_erp_url() 

fetch_url = f"""{url}/api/resource/Cost Center Configuration?filters=[["project","like","Velocity%"]]&limit_page_length=None&fields=["project","`tabCost Center Mapping`.`cost_center`"]""" 

response = session.get(fetch_url) 

projects = {} 

json_obj = [] 

if response.status_code == 200: 

json_obj = json.loads(response.content)['data'] 

fetch_url = f"""{url}/api/resource/Cost Center Configuration?filters=[["project","like","DMR%"]]&limit_page_length=None&fields=["project","`tabCost Center Mapping`.`cost_center`"]""" 

response = session.get(fetch_url) 

erp_cost_centers = [d.name for d in frappe.get_all("ERP Cost Center",filters={})] 

erp_projects = [d.name for d in frappe.get_all("ERP Project",filters={})] 

if response.status_code == 200: 

json_obj.extend(json.loads(response.content)['data']) 

for row in json_obj: 

project = row.get('project') 

cost_center = row.get('cost_center') 

if project not in projects: 

cost_centers = [cost_center] 

projects[project] = cost_centers 

else: 

projects[project].append(cost_center) 

 

for project,cost_centers in projects.items(): 

project_doc = None 

if project in erp_projects: 

project_doc = frappe.get_doc("ERP Project",project) 

else: 

project_doc = frappe.get_doc({"doctype":"ERP Project","project":project,"cost_center":[]}) 

project_doc.insert(True) 

erp_projects.append(project) 

 

if not project_doc.cost_center: 

project_doc.cost_center = [] 

 

project_cost_center = [d.cost_center for d in project_doc.cost_center] 

for cost_center in cost_centers: 

if cost_center not in erp_cost_centers: 

frappe.get_doc({"doctype":"ERP Cost Center","cost_center":cost_center}).save(True) 

erp_cost_centers.append(cost_center) 

if cost_center not in project_cost_center: 

doc = frappe.get_doc({"doctype":"ERP Cost Center Item","cost_center":cost_center,"parent":project_doc.name,"parenttype":"ERP Project","parentfield":"cost_center"}).insert(True) 

project_doc.cost_center.append(doc) 

project_doc.save(True) 

else: 

frappe.log_error(response.content,"erp_project.fetch_erp_project")
