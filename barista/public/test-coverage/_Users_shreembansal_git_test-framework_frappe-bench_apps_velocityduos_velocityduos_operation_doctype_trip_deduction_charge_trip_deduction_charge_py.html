from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

 

from velocityduos.api.utilities import add_communication 

from velocityduos.api.trip_utils import calculate_revenue_details 

 

class TripDeductionCharge(Document): 

def validate(self): 

vendor_amount = self.vendor_amount if isinstance(self.vendor_amount,float) else float(self.vendor_amount) 

client_amount = self.client_amount if isinstance(self.client_amount,float) else float(self.client_amount) 

 

if vendor_amount == 0 and client_amount == 0: 

frappe.throw("Both the Vendor and Client amount Cannot be 0") 

 

if frappe.db.get_value("Lorry Receipt",self.lorry_receipt,"trip") != self.trip: 

frappe.throw(f"Lorry Receipt {self.lorry_receipt} is not Linked to {self.trip}") 

 

def on_update(self): 

if self.workflow_state == "Confirmed by BD": 

if self.vendor_amount < self.client_amount: 

trip_doc = frappe.get_doc("Trip",self.trip) 

if trip_doc.trip_cost and trip_doc.trip_cost.strip(): 

revenue_details_dict = calculate_revenue_details(trip_doc) 

landed_margin=revenue_details_dict['total_revenue']-revenue_details_dict['total_cost'] 

if landed_margin < 0: 

self.workflow_state = "Exceptional Approval Required" 

else: 

self.workflow_state = "Pending Financial Controller Approval" 

else: 

self.workflow_state = "Exceptional Approval Required" 

else: 

self.workflow_state = "Pending Financial Controller Approval" 

self.save() 

add_communication(self.doctype, self.name, self.workflow_state) 

 

def on_trash(self): 

if self.workflow_state in ['Approved','Exceptional Approval Required','Pending Financial Controller Approval'] and 'Developer' not in frappe.get_roles(): 

frappe.throw("Cannot Delete TACs for Approval") 

 

def on_submit(self): 

if self.workflow_state == "Cancelled": 

validate_charge(self) 

if(self.workflow_state == "Approved" or self.workflow_state == "Cancelled"): 

charges = frappe.db.sql_list("""select workflow_state from `tabTrip Additional Charges` where trip=%(trip)s and workflow_state not in ('Cancelled','Approved') UNION ALL select workflow_state from `tabTrip Deduction Charges` where trip=%(trip)s and workflow_state not in ('Cancelled','Approved')""", {'trip': self.trip}) 

if(len(charges) == 0): 

trip = frappe.get_doc("Trip", self.trip) 

if(trip.workflow_state == "Pending Finance Approval"): 

trip.workflow_state = "Additional/Deduction Approved" 

trip.save(True) 

add_communication("Trip", trip.name, trip.workflow_state) 

if self.workflow_state == "Approved" and self.vendor_amount > 0: 

# Create Debit Note for Already Raised Invoices 

vpi = frappe.get_all("Vendor Purchase Invoice",filters=[["Vendor Invoice Item","trip","=",self.trip],["docstatus","!=","2"]],fields=["name"]) 

if len(vpi) >0: 

vpi = frappe.get_doc("Vendor Purchase Invoice",vpi[0].name) 

vpi.create_debit_note() 

 

if self.workflow_state == "Approved" and self.vendor_amount > 0: 

contract_trip = frappe.db.sql_list("""select rr.contract from `tabRoute Rates` rr join `tabTrip` tr on tr.route_rate=rr.name where tr.name=%(trip)s""",{'trip':self.trip}) 

if len(contract_trip) > 0 and (contract_trip[0] and contract_trip[0].strip()): 

contract_tdc = frappe.new_doc("Contract Deduction Charges") 

contract_tdc.contract = contract_trip[0] 

contract_tdc.charge_type = self.description 

contract_tdc.lr = self.lr 

contract_tdc.amount = self.vendor_amount 

contract_tdc.description = self.remarks 

contract_tdc.reference_tdc = self.name 

contract_tdc.insert(True) 

frappe.db.sql("""update `tabContract Deduction Charges` set workflow_state='Approved' where name=%(name)s limit 1""",{'name':contract_tdc.name},auto_commit=1) 

contract_tdc = frappe.get_doc("Contract Deduction Charges",contract_tdc.name) 

contract_tdc.submit()
