from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

 

from frappe.utils import cstr 

from frappe.model.naming import make_autoname 

from frappe import throw, _ 

import re 

 

class Party(Document): 

def autoname(self): 

if self.party_name and self.party_name.strip() and self.address_type: 

self.name = (cstr(self.party_name).strip() + "-" + cstr(_(self.address_type)).strip()) 

if frappe.db.exists("Party", self.name): 

self.name = make_autoname("{0}-.#".format(self.name)) 

else: 

frappe.throw("Please Enter Party Name & Address Type") 

 

def validate(self): 

email_regex = """^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$""" 

if self.postal_code and self.postal_code.strip(): 

if (len(self.postal_code) != 6 or not self.postal_code.isdigit()): 

frappe.throw("Invalid Postal Code") 

if self.email_address and self.email_address.strip and not re.search(email_regex,self.email_address): 

frappe.throw("Invalid Email Address") 

if self.gstn and self.gstn.strip() and len(self.gstn.strip()) != 15: 

frappe.throw("Invalid GSTIN") 

 

def before_save(self): 

full_address = self.party_name 

full_address = f"{full_address},\n{self.address_line1}" 

if self.address_line2 and self.address_line2.strip(): 

full_address = f"{full_address}\n{self.address_line2}" 

full_address = f"{full_address}\n{self.postal_code}" 

full_address = f"{full_address}\n{self.city}" 

full_address = f"{full_address}\n{self.country}\nGSTN:{self.gstn or 'NA'}" 

self.full_address = full_address
