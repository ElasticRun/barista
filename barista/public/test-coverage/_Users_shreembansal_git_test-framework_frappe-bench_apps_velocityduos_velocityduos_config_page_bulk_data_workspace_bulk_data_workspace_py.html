@frappe.whitelist() 

def get_name_list(doctype): 

doc_names = frappe.get_list(doctype, fields=['name'], order_by='name') 

namelist = [rec['name'] for rec in doc_names] 

return namelist 

 

 

@frappe.whitelist() 

def get_all_previous_data_imports(user_email): 

previous_data_import = frappe.get_all('Data Import', filters={'owner': user_email}, fields=[ 

'name as Name', 'reference_doctype as Doctype', 'action as Action', 'import_file as File_Attached', 'import_status as Import_Status', 'modified_by as Last_Modified_By', 'DATE_FORMAT(modified,"%d/%m/%Y") as Modified']) 

previous_child_data_import = frappe.get_all('Data Import For Child', filters={'owner': user_email}, fields=[ 

'name as Name', 'reference_doctype as Doctype', 'action as Action', 'import_file as File_Attached', 'import_status as Import_Status', 'modified_by as Last_Modified_By', 'DATE_FORMAT(modified,"%d/%m/%Y") as Modified']) 

previous_data_import.extend(previous_child_data_import) 

return previous_data_import
