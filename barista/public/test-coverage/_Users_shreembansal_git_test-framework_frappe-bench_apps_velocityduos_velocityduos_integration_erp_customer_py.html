def get_customer_matrix(client_name,session): 

url = get_erp_url()+'api/resource/Customer?filters=[["Customer","customer_name", "=", "{0}" ]]'.format(url_parser.quote(client_name)) 

try: 

if not session or session.get("frappe.auth.get_logged_user").status_code != 200: 

raise frappe.ValidationError("Login ERP Client") 

except Exception: 

session = login_erp_client() 

 

response = session.get(url) 

if (response.status_code == 200): 

json_obj = json.loads(response._content) 

if(len(json_obj['data']) > 0): 

return json_obj['data'][0]['name'] 

else: 

frappe.throw(re.compile('<p>(.*?)</p>', re.DOTALL | re.IGNORECASE).findall(str(response._content)))

def get_item_config(client): 

erp_items = {} 

if frappe.db.exists("ERP Item Mapping",client): 

for item in (frappe.db.sql("""select eit.item,ei.name as erp_mapping,ei.tax_rate_text from `tabERP Mapping Item` eit join `tabERP Item` ei on ei.name=eit.erp_mapping where eit.parent=%(client)s""",{'client':client},as_dict=1)): 

erp_items[item.item] = item 

else: 

for item in (frappe.db.sql("""select act.name,ei.name as erp_mapping,ei.tax_rate_text from `tabAdditional Charge Type` act join `tabERP Item` ei on ei.name=act.erp_mapping""",as_dict=1)): 

erp_items[item.name] = item 

return erp_items 

 

def get_erp_mapping(items,description,standard_item=True): 

default_adhoc_item = "Additional Adhoc Charges" 

default_standard_item = "Transportation Charges" 

item = None 

if description in items: 

item = items.get(description) 

elif standard_item: 

item = items.get(default_standard_item) 

else: 

item = items.get(default_adhoc_item) 

 

return item
