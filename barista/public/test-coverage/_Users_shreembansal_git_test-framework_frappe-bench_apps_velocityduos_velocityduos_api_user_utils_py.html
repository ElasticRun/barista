bd_by_client_zone = """ select distinct pr.user from `tabUser Preference Item` uc  

inner join `tabUser Preference` pr ON uc.parent = pr.name 

inner join tabUser u ON u.name = pr.user  

inner join `tabHas Role` r ON r.parent = u.name 

where uc.customer in %(clients)s  

and uc.territory in %(locations)s  

and pr.enabled = 1 

and r.role = 'Business Development'""" 

 

bd_by_client = """ select distinct pr.user from `tabUser Preference Item` uc  

inner join `tabUser Preference` pr ON uc.parent = pr.name 

inner join tabUser u ON u.name = pr.user  

inner join `tabHas Role` r ON r.parent = u.name

and pr.enabled = 1 

and r.role = 'Business Development'""" 

 

bd = frappe.db.sql(bd_by_client_zone,{"clients":clients,"locations":locations},as_dict = True)

#no bd found 

return None
