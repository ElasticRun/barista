from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

 

from velocityduos.integration.erp_purchase_order import make_matrix_apr 

 

allowed_trip_state = ['In Transit','Awaiting Load','Trip Ended','Advance Released'] 

 

class AdditionalPaymentRequest(Document): 

def before_save(self): 

self.calculate_total_amount() 

 

def on_submit(self): 

pos = [] 

if self.workflow_state == "Approved by Manager": 

if self.additional_payment_item: 

if len(self.additional_payment_item) == 1: 

po = frappe.db.get_value("Trip Additional Charge",self.additional_payment_item[0].additional_charge_id,"erp_purchase_order") 

if po and po.strip(): 

self.erp_purchase_order = po.strip() 

make_matrix_apr(self) 

else: 

raise frappe.ValidationError("Purchase Order is not Raised For TAC") 

elif len(self.additional_payment_item) > 1: 

raise frappe.ValidationError("Only one TAC can be Added per Additional Payment") 

elif len(self.additional_payment_item) == 0: 

raise frappe.ValidationError("No TAC Available For Additional Payment") 

else: 

raise frappe.ValidationError("No TAC Available For Additional Payment") 

 

def calculate_total_amount(self): 

addition_items = [] 

total_amount = 0.0 

trip = frappe.get_doc("Trip",self.trip) 

if(trip.workflow_state not in allowed_trip_state): 

frappe.throw("Trip state is not in either of this status ("+','.join(allowed_trip_state)+")") 

self.validate_trip(trip.route_rate) 

if self.additional_payment_item and len(self.additional_payment_item)>0: 

for items in self.additional_payment_item: 

if items.additional_charge_id in addition_items: 

frappe.throw("Duplicate entries available for ID : " + items.additional_charge_id) 

addition_items.append(items.additional_charge_id) 

if self.trip != items.trip: 

frappe.throw(f"{items.additional_charge_id} is not associated to Trip {self.trip}") 

total_amount += items.vendor_amount 

 

if total_amount <= 0.0: 

frappe.throw("Amount should be greater than 0") 

 

self.validate_apr_no(addition_items) 

self.validate_additional_charge(addition_items) 

max_payable_amount = frappe.get_single("Configuration").max_payable_amount 

max_payable_amount = max_payable_amount if isinstance(max_payable_amount,float) else float(max_payable_amount) 

if(total_amount > max_payable_amount): 

frappe.throw(f"Additional Amount should not be greater than Rs.{max_payable_amount}") 

self.total_amount = total_amount 

else: 

frappe.throw("Enter the Additional Payment Details") 

 

def validate_apr_no(self,addition_items): 

apr_nos = frappe.db.sql_list("""select 

distinct api.additional_charge_id 

from 

`tabAdditional Payment Request Item` api 

join `tabAdditional Payment Request` apr on apr.name = api.parent 

where 

api.additional_charge_id in %(addition_items)s 

and api.parent <> %(apr_no)s 

and apr.workflow_state <> 'Cancelled'""",{'addition_items':addition_items,'apr_no':self.name}) 

if len(apr_nos) >0: 

frappe.throw("Payment already raised for " + ','.join(apr_nos)) 

 

def validate_additional_charge(self,addition_items): 

additional_charges = frappe.get_all("Trip Additional Charge",filters=[["name","in",addition_items]],fields=["*"]) 

for additional_charge in additional_charges: 

if additional_charge.workflow_state != "Approved": 

frappe.throw("TAC {additional_charge.name} is not Approved") 

elif additional_charge.pay_by_cash == 1: 

frappe.throw("TAC {additional_charge.name} is Paid By Cash") 

 

def validate_trip(self,route_rate): 

vendor_pos = frappe.db.sql_list("""SELECT 

pii.trip 

FROM 

`tabPurchase Invoice Item` pii 

join `tabPurchase Invoice` pi on pi.name = pii.parent 

where 

pii.trip = %(trips) s 

and pi.docstatus < 2""",{'trips':self.trip}) 

if len(vendor_pos) >0 : 

frappe.throw(f"Vendor Invoice is already raised for Trip {self.trip}")
