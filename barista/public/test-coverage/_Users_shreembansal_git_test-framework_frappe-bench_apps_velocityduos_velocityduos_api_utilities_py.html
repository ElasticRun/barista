def get_pdf_file(html, file_name, options=None, output=None): 

html = scrub_urls(html) 

html, options = prepare_options(html, options) 

fname = os.path.join("/tmp", "{0}.pdf".format(file_name)) 

 

try: 

pdfkit.from_string(html, fname, options=options or {}) 

except IOError as e: 

if ("ContentNotFoundError" in e.message 

or "ContentOperationNotPermittedError" in e.message 

or "UnknownContentError" in e.message 

or "RemoteHostClosedError" in e.message): 

pass 

else: 

print("Get PDF File" + str(e)) 

 

return fname

def is_tds_applicable(transporter,check_date=None): 

is_tds_applicable = True 

dates = get_financial_year_dates(check_date) 

financial_year = "{0}-{1}".format(dates[0].strftime("%Y"),dates[1].strftime("%y")) 

tds_declaration = frappe.get_all("Tds Declaration",filters={'parent':transporter,'financial_year':financial_year,'status':'Active','has_more_than_10_fleet':('!=',1)},fields=["*"]) 

if len(tds_declaration) > 0: 

if not tds_declaration[0].has_more_than_10_fleet and tds_declaration[0].declaration_certificate and tds_declaration[0].declaration_certificate.strip(): 

trips = frappe.db.sql_list("""select distinct tr.vehicle  

from `tabTrip` tr join `tabTrip Cost` tc  

on tc.name=tr.trip_cost  

where tc.transporter=%(transporter)s  

and tr.creation >= %(start)s  

and tr.creation <= %(end)s  

and tr.vehicle is not NULL""", 

{ 

'transporter':transporter, 

'start':dates[0], 

'end':dates[1] 

} 

) 

if len(trips) > 10: 

is_tds_applicable = True 

frappe.db.set_value("Tds Declaration", tds_declaration[0].name, "has_more_than_10_fleet", 1) 

else: 

is_tds_applicable = False 

return is_tds_applicable 

 

def get_financial_year_dates(dt=None): 

"""Returns start date and end date of the finantial year for given date 

if no date is given, current finantial year will be returns 

 

Parameters 

---------- 

dt: datetime 

 

Returns 

------- 

tuple 

tuple consisting start_date and end_date 

""" 

if dt is None: 

dt = frappe.utils.now_datetime().date() 

if isinstance(dt, str): 

dt = frappe.utils.get_datetime(dt).date() 

 

start_date = None 

end_date = None 

 

if dt.month > 3: 

start_date = frappe.utils.datetime.date(dt.year, 4, 1) 

end_date = frappe.utils.datetime.date(dt.year + 1, 3, 31) 

else: 

start_date = frappe.utils.datetime.date(dt.year - 1, 4, 1) 

end_date = frappe.utils.datetime.date(dt.year, 3, 31) 

 

return start_date, end_date

@frappe.whitelist() 

def send_email(self,method): 

if self.reference_doctype == "Lorry Receipt" and self.communication_type == "Comment" and self.comment_type == "Comment": 

subject = "Lorry Receipt {lr} - Comments Added by {sender}".format(lr=self.reference_name,sender=self.sender) 

message = "{sender} has added the following comments in Lorry Receipt <a href='/desk#Form/Lorry%20Receipt/{lr}'>{lr}</a><br/><br/>{content}".format(lr=self.reference_name,sender=self.sender_full_name,content=self.subject) 

 

lorry_receipt = frappe.get_doc("Lorry Receipt",self.reference_name) 

if lorry_receipt.trip is not None and lorry_receipt.trip.strip() != "": 

trips = frappe.get_doc("Trip",lorry_receipt.trip) 

consignment = frappe.get_doc("Consignment",lorry_receipt.consignment) 

 

# Get the Recipients 

recipients = [] 

if consignment.bd: 

recipients.append(consignment.bd) 

if trips.vm: 

recipients.append(trips.vm) 

if lorry_receipt.consignor_ground_ops: 

recipients.append(lorry_receipt.consignor_ground_ops) 

if lorry_receipt.consignee_ground_ops: 

recipients.append(lorry_receipt.consignee_ground_ops) 

# Get the Cc List 

cc=frappe.db.sql_list("""select 

distinct uc.user 

from 

`tabHas Role` hr 

inner join `tabUser Preference` uc on hr.parent = uc.user 

where 

hr.role = 'Finance Controller' 

and uc.all_pref = 1 

and uc.enabled = 1 

union 

all 

select 

z.territory_owner 

from 

`tabTerritory` z 

where 

z.name in ('{origin_zone}', '{dest_zone}')""".format(origin_zone=consignment.origin_territory,dest_zone=consignment.destination_territory)) 

cc.append(self.sender) 

# Send the mail 

recipients = get_active_mail_recipients(recipients) 

cc = get_active_mail_recipients(cc) 

frappe.sendmail(recipients=recipients,subject=subject,message=message,reference_doctype=self.doctype,reference_name=self.name,cc=cc,expose_recipients='header') 

# Send Push Notification 

message = "New Message for {0}".format(self.reference_name) 

for email in recipients: 

push_notification("INFO", message, email,"CHAT",self.reference_name,"lr_messages")
