548def calculate_distance(pickup_lat,pickup_lng,delivery_lat,delivery_lng): 

549 from math import sin, cos, sqrt, atan2, radians 

550 distance = 0.0 

551 try: 

552 R = 6373.0 

553 lat1 = radians(float(pickup_lat)) 

554 lon1 = radians(float(pickup_lng)) 

555 lat2 = radians(float(delivery_lat)) 

556 lon2 = radians(float(delivery_lng)) 

557 dlon = lon2 - lon1 

558 dlat = lat2 - lat1 

559 a = sin(dlat / 2)**2 + cos(lat1) * cos(lat2) * sin(dlon / 2)**2 

560 c = 2 * atan2(sqrt(a), sqrt(1 - a)) 

561 distance = R * c
