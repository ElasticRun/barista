369 def validate_contract_shipping_price(self): 

370 if self.loads and len(self.loads) > 0: 

371 trip_types = [] 

372 freight_types = [] 

373 lane = frappe.get_doc("Lane",self.loads[0].lane) 

374 for load in self.loads: 

375 trip_types.append(load.trip_type) 

376 freight_types.append(load.freight_type) 

377 

378 trip_types = list(set(trip_types)) 

379 freight_types = list(set(freight_types)) 

380 

381 if len(trip_types) > 1: 

382 frappe.throw("Cannot combine Contract & Spot Type in same Trip") 

383 

384 if trip_types[0] == 'Contract' and 'Full Freight' in freight_types: 

385 contract_price = frappe.db.sql_list("""SELECT cpi.rate 

386 FROM `tabContract Pricing Item` cpi join `tabCompany Contract` cc  

387 on cc.name=cpi.parent 

388 join `tabCompany` c  

389 on c.contract=cc.name 

390 WHERE c.name = %(client)s  

391 AND cc.status='Active'  

392 AND ((cpi.vehicle_type=%(vehicle_type)s AND cc.applicable_for_all_lanes) OR (cpi.vehicle_type=%(vehicle_type)s 

393 AND cpi.start_city=%(origin_city)s 

394 AND cpi.end_city=%(destination_city)s))""" 

395 ,{"client":self.loads[0].customer, 

396 "vehicle_type":self.loads[0].vehicle_type, 

397 "origin_city":lane.start_location, 

398 "destination_city":lane.end_location} 

399 ) 

400 

401 if len(contract_price) == 0: 

402 frappe.throw("No Contract Price Available for selected Lane & Vehicle Type {0}-{1}-{2}".format(lane.start_location,lane.end_location,self.loads[0].vehicle_type)) 

403 if type(contract_price[0]) is str: 

404 contract_price = contract_price[0].replace(',','').replace(' ','') 

405 else: 

406 contract_price = contract_price[0] 

407 

408 contract_price = contract_price if type(contract_price) is float else float(contract_price) 

409 message = "<table class='table'><tr><th>LR</th><th>Amount</th></tr>" 

410 

411 total_freight = 0.0 

412 for load in self.loads: 

413 if load.trip_type == 'Contract' and load.freight_type == 'Full Freight': 

414 shipping_price = load.shipping_price if type(load.shipping_price) is float else float(load.shipping_price) 

415 total_freight += shipping_price 

416 message = message + "<tr><td>{0}</td><td>{1}</td></tr>".format(load.name,load.shipping_price) 

417 message = message + "</table>" 

418 

419 if total_freight != contract_price: 

420 frappe.throw("Total Freight should be equal to Contract Price <b>{0}</b>. Please redistribute the Freight. <br>{1}".format(contract_price,message))
