from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

 

from velocityduos.integration.erp_purchase_order import create_additional_po,create_purchase_order 

from velocityduos.integration.erp_sales_order import create_additional_so,create_sales_order 

from velocityduos.integration.erp_purchase_receipt import create_purchase_receipt 

from velocityduos.integration.erp_delivery_note import create_delivery_note 

 

from velocityduos.integration.erp_client_login import login_erp_client,get_erp_url 

 

class ERPSyncQueue(Document): 

def before_save(self): 

try: 

if self.sync_status == "Failed" and self.get_doc_before_save().sync_status != self.sync_status: 

recipients = frappe.get_single("Configuration").support_emails 

if recipients: 

recipients = recipients.split(",") 

subject = "IMPORTANT! - Matrix {0} Sync Failed for {1} {2}".format(self.erp_doctype,self.source_type,self.source_document) 

message = "Matrix Sync has Failed for <a style=\"color:blue\" href=\"/desk#Form/Matrix%20Sync%20Queue/{0}\">{1}</a><br><br> <b>Logs:-</b><br><br>{2}".format(self.name,self.source_document,self.logs) 

frappe.sendmail(recipients=recipients,subject=subject,message=message,reference_doctype=self.doctype,reference_name=self.name) 

elif self.sync_status == "Success": 

self.logs = None 

except Exception: 

frappe.log_error(frappe.get_traceback(),"matrix_sync_queue-before_save-{0}".format(self.name)) 

 

def job_sync_queue(): 

session = None 

session_header = None 

try: 

session = login_erp_client() 

session_header = session.headers 

except Exception: 

recipients = frappe.get_single("Configuration").support_emails 

if recipients: 

recipients = recipients.split(",") 

subject = "IMPORTANT! - Matrix Connectivity Issue" 

message = "Matrix Sync Job has Not Been Intitated as Matrix is unreachable. <br><br>Please find below error details. <br><br>{0}".format(frappe.get_traceback()) 

frappe.sendmail(recipients=recipients,subject=subject,message=message) 

return 

 

sync_docs = frappe.db.sql_list("""select 

name 

from 

`tabERP Sync Queue` 

where 

sync_status = %(status) s 

order by 

FIELD( 

erp_doctype,'Purchase Order','Sales Order','Purchase Receipt', 

'Delivery Note','Purchase Invoice','Additional Purchase Invoice', 

'Purchase Invoice Return','Sales Invoice','Additional Sales Invoice', 

'Sales Invoice Return','Journal Entry' 

) ASC""", 

{ 

'status':'Pending' 

} 

) 

 

if len(sync_docs) > 0: 

frappe.db.sql("""update `tabERP Sync Queue` set sync_status='In Progress' where name in %(name)s""",{'name':sync_docs},auto_commit=1) 

for row in sync_docs: 

sync_doc = frappe.get_doc("ERP Sync Queue",row) 

try: 

if not sync_doc.erp_reference_name or not sync_doc.erp_reference_name.strip(): 

session.headers = session_header 

process_doc(sync_doc,session) 

except Exception: 

sync_doc.logs = frappe.get_traceback() 

sync_doc.sync_status = "Failed" 

sync_doc.save(True) 

frappe.db.commit() 

 

def process_doc(sync_doc, session=None): 

if not session: 

session = login_erp_client() 

 

if sync_doc.erp_doctype == "Purchase Order": 

if sync_doc.source_type == "Trip Additional Charge": 

create_additional_po(sync_doc,session) 

elif sync_doc.source_type == "Purchase Order": 

create_purchase_order(sync_doc) 

elif sync_doc.erp_doctype == "Sales Order": 

if sync_doc.source_type == "Trip Additional Charge": 

create_additional_so(sync_doc,session) 

elif sync_doc.source_type == "Sales Order": 

create_sales_order(sync_doc,session) 

else: 

raise frappe.ValidationError("Invalid ERP Sync DocType") 

elif sync_doc.erp_doctype == "Purchase Receipt": 

if sync_doc.source_type in ["Trip Additional Charge","Purchase Order"]: 

create_purchase_receipt(sync_doc,session) 

elif sync_doc.erp_doctype == "Delivery Note": 

if sync_doc.source_type in ["Trip Additional Charge","Sales Order"]: 

create_delivery_note(sync_doc,session)
