def update_payment_voucher_po(trip_doc,po,session): 

pv_doc = frappe.get_doc("ERP Payment Voucher",trip_doc.payment_voucher) 

if pv_doc.received_from == 'Company': 

return 

url = f'{get_erp_url()}api/resource/Payment Voucher/{pv_doc.matrix_voucher_ref_no}' 

response = session.get(url) 

if(response.status_code == 200): 

json_obj = json.loads(response._content)['data'] 

if("purchase_order" in json_obj and json_obj["purchase_order"].strip() != ""): 

pass 

else: 

name = json_obj['name'] 

json_obj['purchase_order'] = po 

url = f'{get_erp_url()}api/resource/Payment Voucher/{name}' 

response = session.put(url,json.dumps(json_obj)) 

if(response.status_code == 200): 

json_obj = json.loads(response._content) 

update_doc = json_obj['data'] 

pi_doc = {'doc':json.dumps(update_doc),'action':'Submit PO Voucher for Approval','cmd':'frappe.model.workflow.apply_workflow'} 

response = session.post(url,pi_doc) 

json_obj = json.loads(response._content) 

update_doc = json_obj['message'] 

pi_doc = {'doc':json.dumps(update_doc),'action':'Approve','cmd':'frappe.model.workflow.apply_workflow'} 

response = session.post(url,pi_doc) 

pv_doc.erp_purchase_order = po 

pv_doc.save(True) 

 

def get_tds_amount(pi_num,session=None): 

tds_amount = 0.0 

try: 

vpi_url = f"{get_erp_url()}api/resource/Purchase Invoice" 

vpi_filters = '?filters={"name":"'+pi_num+'"}&fields=["base_taxes_and_charges_deducted"]' 

if not session: 

session = login_erp_client() 

response = session.get(vpi_url+""+vpi_filters) 

if(response.status_code == 200): 

json_obj = json.loads(response._content) 

list = json_obj['data'] 

if(len(list)>0): 

tds_amount=list[0]['base_taxes_and_charges_deducted'] 

except Exception as e: 

frappe.log_error(str(e),"Velocity") 

return tds_amount
