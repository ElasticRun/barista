from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

from datetime import date 

 

class GoalTemplate(Document): 

def autoname(self): 

self.name=set_goal_template_name(self) 

 

def validate(self): 

if self.start_date>self.end_date: 

frappe.throw('Start Date cannot be greater than End Date') 

 

total_weightage=0 

goal_dct={} 

for item in self.goal_template_items: 

total_weightage+=float(item.weightage) 

if item.goal not in goal_dct: 

goal_dct[item.goal]=1 

else: 

frappe.throw('Goal- {goal} is already added. Unique Goals need to be added in the template.'.format(goal=item.goal)) 

 

if total_weightage!=100: 

frappe.throw('Total Weightage should be 100%'+'<br>Current Weightage is '+str(total_weightage)+'%') 

 

overlaped_templates=frappe.db.sql("""select name,role,start_date,end_date from `tabGoal Template` where name!=%(name)s and role=%(role)s and ((%(start_date)s between start_date and end_date) or (%(end_date)s between start_date and end_date)) #and workflow_state in ('Submitted','Initiated')""",{'role':self.role,'start_date':self.start_date,'end_date':self.end_date,'name':self.name},as_dict=1,debug=0) 

if len(overlaped_templates): 

frappe.throw('Already Goal Template- {name} for this Designation {role} with Start Date {start_date} and End Date {end_date} overlapping with current Goal Template exists'.format(name=overlaped_templates[0]['name'],role=overlaped_templates[0]['role'],start_date=overlaped_templates[0]['start_date'],end_date=overlaped_templates[0]['end_date'])) 

 

 

@frappe.whitelist() 

# velocityduos.pms.doctype.goal_template.goal_template.rename_goal_template 

def rename_goal_template(old_name): 

if frappe.db.exists('Goal Template',old_name): 

self=frappe.get_doc('Goal Template',old_name) 

new_name=set_goal_template_name(self) 

if self.name!=new_name: 

frappe.rename_doc(self.doctype,self.name,new_name,ignore_permissions=1,force=1) 

frappe.db.commit() 

return new_name 

 

 

def set_goal_template_name(self): 

role_words=self.role.split(' ') 

role_first_letter=[word[0] for word in role_words] 

role_name=''.join(role_first_letter) 

return role_name+'/'+str(self.start_date)+'/'+str(self.end_date)

return frappe.get_all('Business Hierarchy',filters=[['parent_business_hierarchy','=',designation],['is_group','=',0]],fields=['*'])
