597def get_best_matching_lane(origin_city, destination_city, trip_lanes, matching_threshold): 

598 best_matches = {"source_name": "", "source_score": 0, "destination_name": "", "destination_score": 0} 

599 for j in range(0, len(trip_lanes)): 

600 try: 

601 source_match_score = SequenceMatcher(None, origin_city.lower(), trip_lanes[j]["full_source_name"].lower()).ratio() 

602 destination_match_score = SequenceMatcher(None, destination_city.lower(), trip_lanes[j]["full_destination_name"].lower()).ratio() 

603 except AttributeError: 

604 source_match_score = 0 

605 destination_match_score = 0 

606 if source_match_score >= best_matches["source_score"] and destination_match_score >= best_matches["destination_score"]: 

607 best_matches["source_score"] = source_match_score 

608 best_matches["destination_score"] = destination_match_score 

609 best_matches["source_name"] = trip_lanes[j]["full_source_name"] 

610 best_matches["destination_name"] = trip_lanes[j]["full_destination_name"] 

611 return best_matches
