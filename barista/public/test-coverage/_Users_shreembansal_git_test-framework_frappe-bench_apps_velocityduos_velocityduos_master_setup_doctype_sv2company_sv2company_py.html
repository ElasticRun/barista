from __future__ import unicode_literals 

import frappe 

import re 

from frappe.model.document import Document 

 

pan_format = """[a-zA-Z]{5}[0-9]{4}[a-zA-Z]{1}""" 

email_regex = """^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,3})+$""" 

 

class sv2Company(Document): 

def validate(self): 

if not self.is_parent and not self.parent_company: 

frappe.throw("Please Map Parent Holding Company or Mark the Company as Parent") 

if not self.is_parent and frappe.db.get_value("sv2Company",self.parent_company,"status") != 'Active': 

frappe.throw("Parent Holding Company cannot be Inactive") 

if self.pan and len(self.pan.strip()) != 10 and len(re.findall(pan_format,self.pan_number)) == 0: 

frappe.throw("Not a Valid PAN") 

if self.communication_email and self.communication_email.strip() and not re.search(email_regex,self.communication_email): 

frappe.throw("Not a Valid Communication Email Address")
