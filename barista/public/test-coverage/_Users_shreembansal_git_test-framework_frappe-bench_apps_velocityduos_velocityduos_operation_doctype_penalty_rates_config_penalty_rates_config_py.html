from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

from datetime import datetime 

 

 

class PenaltyRatesConfig(Document): 

def before_insert(self): 

self.name = self.penalty_type 

 

def validate(self): 

active_penalty_for_penalty_type = frappe.get_all('Penalty Rates Config', filters={ 

'workflow_state': 'Active', 'penalty_type': self.penalty_type}, fields=['name']) 

 

if len(active_penalty_for_penalty_type): 

# lst = [dct['name'] for dct in active_penalty_for_penalty_type] 

frappe.throw('Already Penalty Rates are configured in Active state for this Penalty Type' + 

'<br>'+'Please Deactivate existing Penalty Rates and create new') 

 

if self.penalty_type == 'POD Lost': 

self.flags.ignore_mandatory = True 

 

if not self.fixed_amount or self.fixed_amount <= 0: 

frappe.throw( 

'Please provide Fixed Amount for this Penalty Type')

def process_penalty(lr_doc=None, incident_doc=None): 

try: 

if lr_doc and not incident_doc and lr_doc.pod_lost != 1: 

if lr_doc.workflow_state == 'POD Collected by GOPS': 

 

dt = lr_doc.vehicle_release_time 

if isinstance(dt, str): 

dt = frappe.utils.get_datetime(dt).date() 

else: 

dt = lr_doc.vehicle_release_time.date() 

penalty_days = datetime.now().date()-dt 

penalty_days = penalty_days.days 

 

penalty_amount = calculate_pod_late_submission_penalty( 

penalty_days) 

if penalty_amount > 0: 

create_tdc(lr_doc.name, lr_doc.trip, 'POD Late Submission', penalty_amount, 

'Vehicle Release Date {0}\n POD Collected by GOPS Date {1}\n No Days between vehicle release date and POD submited date {2}'.format(str(dt), str(datetime.now().date()), penalty_days)) 

elif not lr_doc and incident_doc: 

penalty_amount = calculate_pod_lost_penalty() 

lr_doc = frappe.get_doc("Lorry Receipt", incident_doc.entity_id) 

if penalty_amount > 0 and lr_doc.pod_lost == 1: 

create_tdc(lr_doc.name, lr_doc.trip, 'POD Lost', 

penalty_amount, 'POD Lost Penalty') 

except Exception as ex: 

print('ex', ex) 

frappe.log_error(frappe.get_traceback(), 

'PODPenaltyRates|process_penalty') 

 

 

def calculate_pod_lost_penalty(): 

penalty_amount = 0 

try: 

pl_doc = frappe.get_all('Penalty Rates Config', filters={ 

'workflow_state': 'Active', 'penalty_type': 'POD Lost'}, fields=['name']) 

 

if len(pl_doc): 

pl_doc = frappe.get_doc('Penalty Rates Config', pl_doc[0]['name']) 

penalty_amount = pl_doc.fixed_amount 

except Exception: 

frappe.log_error(frappe.get_traceback(), 

'PODPenaltyRates|calculate_pod_lost_penalty') 

return penalty_amount 

 

 

def calculate_pod_late_submission_penalty(penalty_days): 

penalty_amount = 0 

try: 

pls_doc = frappe.get_all('Penalty Rates Config', filters={ 

'workflow_state': 'Active', 'penalty_type': 'POD Late Submission'}, fields=['name']) 

 

if len(pls_doc): 

pls_doc = frappe.get_doc( 

'Penalty Rates Config', pls_doc[0]['name']) 

penalty_amount = process_slab_rates( 

pls_doc.slab_rates, penalty_days) 

except Exception: 

frappe.log_error(frappe.get_traceback( 

), 'PenaltyRatesConfig|calculate_pod_late_submission_penalty') 

 

return penalty_amount 

 

 

def process_slab_rates(slab_rates, penalty_days): 

penalty_amount = 0 

try: 

for rate in slab_rates: 

rate.value_from = int(rate.value_from or 0) 

rate.value_to = int(rate.value_to or 0) 

# print('pa-',penalty_amount) 

# print('op-',rate.operator) 

if rate.operator == 'Between': 

if (penalty_days >= rate.value_from) and (penalty_days <= rate.value_to): 

penalty_amount = rate.amount 

elif rate.operator == 'Equal to': 

if (penalty_days == rate.value_from) or (penalty_days == rate.value_to): 

penalty_amount = rate.amount 

elif rate.operator in ['>=', '&gt;=']: 

if penalty_days >= rate.value_from: 

penalty_amount = rate.amount 

elif rate.operator in ['>', '&gt;']: 

if penalty_days > rate.value_from: 

penalty_amount = rate.amount 

elif rate.operator in ['<=', '&lt;=']: 

if penalty_days <= rate.value_from: 

penalty_amount = rate.amount 

elif rate.operator in ['<', '&lt;']: 

if penalty_days < rate.value_from: 

penalty_amount = rate.amount 

except Exception: 

frappe.log_error(frappe.get_traceback(), 

'PenaltyRatesConfig|process_slab_rates') 

return penalty_amount 

 

 

def create_tdc(lr, trip, penalty_type, penalty_amount, remarks): 

try: 

existing_tdc = frappe.get_all('Trip Deduction Charges', filters={ 

'source': 'System', 'description': penalty_type, 'trip': trip}) 

if len(existing_tdc) > 0: 

return 

tdc_amts = frappe.db.sql_list( 

"""select sum(vendor_amount) from `tabTrip Deduction Charges` where trip = %(trip)s""", {'trip': trip}) 

print(tdc_amts) 

tdc_total = 0.0 

if tdc_amts and len(tdc_amts) != 0: 

if tdc_amts[0] is not None: 

tdc_total = tdc_amts[0] 

trip_doc = frappe.get_doc("Trip", trip) 

if trip_doc.skip_po == 1: 

return 

route_rate = frappe.get_doc("Route Rates", trip_doc.route_rate) 

 

if route_rate.contract is not None and route_rate.contract.strip() != '': 

return 

 

tdc_total = (route_rate.total_charges - 

route_rate.advance_amount-tdc_total) 

 

if penalty_amount <= tdc_total: 

penalty_amount = penalty_amount 

else: 

penalty_amount = tdc_total 

 

if len(existing_tdc) == 0 and penalty_amount > 0: 

tdc_doc = frappe.new_doc('Trip Deduction Charges') 

tdc_doc.lr = lr 

tdc_doc.trip = trip 

tdc_doc.description = penalty_type 

tdc_doc.vendor_amount = penalty_amount 

tdc_doc.source = 'System' 

tdc_doc.remarks = remarks 

tdc_doc.insert(True) 

frappe.db.set_value(tdc_doc.doctype, tdc_doc.name, 

"workflow_state", "Pending Financial Controller Approval") 

except Exception: 

frappe.log_error(frappe.get_traceback(), 'PODPenaltyRates|create_tdc')
