import frappe 

import json 

from datetime import timedelta,datetime 

import dateutil.parser 

 

from velocityduos.api.utilities import get_financial_year_dates 

from velocityduos.api.mobile_utils import get_app_version 

 

source_workflow_states = ['Vehicle Placed Loading','Documents Uploaded','Documents Rejected','Documents Verified','LR Completed'] 

destination_workflow_states = ['Departed Origin','Pending FOD Confirmation','FOD Confirmed by Client','FOD Verified','Arrived at Destination','POD Collected by GOPS','POD Collected by Driver','POD Soft Copy Verified'] 

 

@frappe.whitelist() 

def get_vendor_actions_count(vendor_name): 

vpo_count_lst = frappe.db.sql_list("""select 

ifnull(count(pi.name), 0) as 'count' 

from 

`tabPurchase Invoice` pi 

where 

pi.bill_from = %(vendor_name)s 

and pi.workflow_state = %(workflow)s""",{'vendor_name':vendor_name,'workflow':'Vendor Acceptance Required'}) 

vpo_count = vpo_count_lst[0] 

 

thc_list = frappe.db.sql_list("""select 

ifnull(count(thc.name), 0) as count 

from 

`tabTrip Hiring Contract` thc 

where 

thc.transporter = %(vendor_name)s 

and thc.workflow_state = %(workflow)s""",{'workflow':'Pending Vendor Acceptance','vendor_name':vendor_name}) 

thc_count = thc_list[0] 

 

result = {} 

result['invoice_count'] = vpo_count 

result['thc_count'] = thc_count 

result['vendor_name'] = vendor_name 

result['version'] = get_app_version() 

# LR Details for Variable GOPS 

vendor_doc = frappe.get_doc("Transporter",vendor_name) 

dates = get_financial_year_dates() 

result['financial_year'] = "{0}-{1}".format(dates[0].strftime('%Y'),dates[1].strftime('%y')) 

tds_declaration = frappe.get_all("Tds Declaration",filters={'parent':vendor_doc.name,'status':'Active','has_more_than_10_fleet':('!=',1),'financial_year':result['financial_year']},fields=["*"]) 

if len(tds_declaration) > 0: 

if not tds_declaration[0].declaration_certificate or not tds_declaration[0].declaration_certificate.strip(): 

result['tds_declaration'] = "false" 

result['tds_declaration_name'] = tds_declaration[0].name 

else: 

result['tds_declaration'] = "true" 

result['tds_declaration_name'] = tds_declaration[0].name 

 

if vendor_doc.type == 'Variable GOPS': 

result['variable_gops'] = True 

result['roles'] = frappe.get_roles(vendor_doc.email_address) 

result['email_id'] = vendor_doc.email_address 

result["pending_lr_completion"] = frappe.db.count("Lorry Receipt",{"workflow_state":("in",source_workflow_states),"consignor_ground_ops":vendor_doc.email_id}) 

result["pending_pod_completion"] = frappe.db.count("Lorry Receipt",{"workflow_state":("in",destination_workflow_states),"consignee_ground_ops":vendor_doc.email_id}) 

result["contact_person"] = vendor_doc.vendor_owner 

if vendor_doc.vendor_owner: 

result["mobile_number"] = frappe.db.get_value("User",vendor_doc.vendor_owner,"mobile_no") 

else: 

result['variable_gops'] = False 

result['email_id'] = vendor_doc.email_address 

result["pending_lr_completion"] = 0 

result["pending_pod_completion"] = 0 

return result 

 

@frappe.whitelist() 

def print_thc(thc_no,lang="English"): 

result = {} 

html = frappe.get_print("Trip Hiring Contract",thc_no,"THC Mobile Format",doc=None,no_letterhead=0) 

thc_tc = frappe.get_doc("THC Terms and Conditions",lang) 

html = html.replace("******THCTC******",thc_tc.terms) 

result['content'] = html 

return result

@frappe.whitelist() 

def get_lr_details(source_variable_gops=None,destination_variable_gops=None): 

if source_variable_gops and source_variable_gops.strip(): 

result = frappe.db.sql("""select lr.*, 

if(ttl.consent_status = "ALLOWED", True, False) as is_sim, 

if(ftt.driverapp_count > 0, True, False) as is_app, 

tr.driver_phone_no, 

tr.driver 

from `tabLorry Receipt` lr  

join `tabTrip` tr on lr.trip=tr.name 

left join `tabTrip Tracking Log` ttl on ttl.name=lr.trip  

left join 

(select ft.trip_no,  

sum(case when source='Driver App' then 1 else 0 end) as 'driverapp_count'  

from `tabFleet Tracker` ft where docstatus<> 2 group by ft.trip_no) ftt on ftt.trip_no=lr.trip 

where lr.source_variable_gops=%(gops)s 

and lr.workflow_state in %(workflow)s""",{'gops':source_variable_gops,'workflow':source_workflow_states} 

,as_dict=1) 

elif destination_variable_gops and destination_variable_gops.strip(): 

result = frappe.db.sql("""select lr.*, 

if(ttl.consent_status = "ALLOWED", True, False) as is_sim, 

if(ftt.driverapp_count > 0, True, False) as is_app, 

tr.driver_phone_no, 

tr.driver 

from `tabLorry Receipt` lr join `tabTrip` tr on lr.trip=tr.name 

left join `tabTrip Tracking Log` ttl on ttl.name=lr.trip  

left join 

(select ft.trip_no,  

sum(case when source='Driver App' then 1 else 0 end) as 'driverapp_count'  

from `tabFleet Tracker` ft where docstatus<> 2 group by ft.trip_no) ftt on ftt.trip_no=lr.trip 

where lr.destination_variable_gops=%(gops)s 

and lr.workflow_state in %(workflow)s""",{'gops':destination_variable_gops,'workflow':destination_workflow_states} 

,as_dict=1) 

 

if(result): 

transition = frappe.get_all('Workflow', filters={"document_type":"Lorry Receipt", "is_active":1},fields=['*']) 

action_map = {} 

next_state_map = {} 

if(transition): 

trans_rules = frappe.get_doc("Workflow",transition[0].workflow_name) 

for trans in trans_rules.transitions: 

if(trans.state in action_map): 

arr = action_map[trans.state] 

if(trans.action in arr): 

pass 

else: 

arr.append(trans.action) 

action_map[trans.state] = arr 

else: 

arr = [] 

arr.append(trans.action) 

action_map[trans.state] = arr 

 

if(trans.state in next_state_map): 

arr = next_state_map[trans.state] 

if(trans.next_state in arr): 

pass 

else: 

arr.append(trans.next_state) 

next_state_map[trans.state] = arr 

else: 

arr = [] 

arr.append(trans.next_state) 

next_state_map[trans.state] = arr 

 

for value in result: 

if(value.workflow_state in action_map): 

value.action = action_map[value.workflow_state] 

if(value.workflow_state in next_state_map): 

value.next_state = next_state_map[value.workflow_state] 

 

return result 

 

@frappe.whitelist() 

def update_person_info(mobile_no,imei): 

status_code = {"valid_vendor":False} 

try: 

person = frappe.get_all("Person",filters={"contact_number":mobile_no},fields=["name","enforce_restriction"]) 

if person and len(person)>0: 

# person = frappe.get_doc("Person",person[0]) 

person = person[0] 

if person.enforce_restriction == 1: 

available_imeis = frappe.db.sql_list("""select parent from `tabImei Detail` where imei_number=%(imei)s""",{'imei':imei}) 

if len(available_imeis) > 0: 

if person.name == available_imeis[0]: 

status_code['valid_vendor'] = True 

else: 

person = frappe.get_doc("Person",person.name) 

imei_detail = frappe.new_doc("Imei Detail") 

imei_detail.imei_number = imei 

imei_detail.mobile_number = mobile_no 

imei_detail.parentfield = "imei_details" 

person.imei_details.append(imei_detail) 

person.save(True) 

frappe.db.commit() 

status_code['valid_vendor'] = True 

else: 

status_code['valid_vendor'] = False 

except Exception: 

status_code['valid_vendor'] = False 

frappe.log_error(frappe.get_traceback(),"velocity-update_person_info") 

 

return status_code
